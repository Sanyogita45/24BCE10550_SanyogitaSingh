Q1) List the Employee Name, their Department Name, and the Location Name for all employees.

SELECT e.employee_name, d.dept_name, d.location_id
FROM Employees e
JOIN Departments d ON e.dept_id = d.dept_id;



Q3) Find the names of Employees who have worked on a project with Client Name 'GlobalTech Solutions'. List the Employee Name and Project Name.

SELECT e.employee_name, p.project_name
FROM Employees e
JOIN Assignments a ON e.employee_id = a.employee_id
JOIN Projects p ON a.project_id = p.project_id
WHERE p.client_name = 'GlobalTech Solutions';



Q4) List all Departments that have no Employees currently assigned to them.

SELECT d.dept_name
FROM Departments d
LEFT JOIN Employees e ON d.dept_id = e.dept_id
WHERE e.employee_id IS NULL;




Q6) Calculate the Average Hours Worked per Employee across all projects. List the Employee Name and the average hours.

SELECT e.employee_name, AVG(a.hours_worked) AS avg_hours
FROM Employees e
JOIN Assignments a ON e.employee_id = a.employee_id
GROUP BY e.employee_name;




Q7) Find the Department Name(s) that have an average salary above the company-wide average salary.

SELECT d.dept_name, AVG(e.salary) AS avg_salary
FROM Employees e
JOIN Departments d ON e.dept_id = d.dept_id
GROUP BY d.dept_name
HAVING AVG(e.salary) > (SELECT AVG(salary) FROM Employees);




Q8) Identify the Client Name(s) whose projects have a total combined hours worked exceeding 500 hours.

SELECT p.client_name, SUM(a.hours_worked) AS total_hours
FROM Projects p
JOIN Assignments a ON p.project_id = a.project_id
GROUP BY p.client_name
HAVING SUM(a.hours_worked) > 500;




Q10) Calculate the total number of employees assigned to each Project, but only include projects that have more than 5 employees assigned.

SELECT p.project_name, COUNT(a.employee_id) AS total_employees
FROM Projects p
JOIN Assignments a ON p.project_id = a.project_id
GROUP BY p.project_name
HAVING COUNT(a.employee_id) > 5;





Q12) Retrieve the Department Name(s) that have employees assigned to more than one project.


SELECT d.dept_name
FROM Departments d
JOIN Employees e ON d.dept_id = e.dept_id
JOIN Assignments a ON e.employee_id = a.employee_id
GROUP BY d.dept_name, e.employee_id
HAVING COUNT(DISTINCT a.project_id) > 1;





Q13) Write a query to list the Employee Name and Project Name for any employee who has worked less than 10 hours on a project.

SELECT e.employee_name, p.project_name, a.hours_worked
FROM Employees e
JOIN Assignments a ON e.employee_id = a.employee_id
JOIN Projects p ON a.project_id = p.project_id
WHERE a.hours_worked < 10;




Q14) Find the Project Name(s) that have no assigned employees.

SELECT p.project_name
FROM Projects p
LEFT JOIN Assignments a ON p.project_id = a.project_id
WHERE a.assignment_id IS NULL;





Q16) Assuming the Assignments table has a start_date column, find the Employee Name and their oldest project start date.

SELECT e.employee_name, MIN(a.start_date) AS oldest_start_date
FROM Employees e
JOIN Assignments a ON e.employee_id = a.employee_id
GROUP BY e.employee_name;





Q17) Write a MySQL query using the DATE_FORMAT function to list all projects that were active in the year 2023.

SELECT project_name, DATE_FORMAT(start_date, '%Y-%m-%d') AS formatted_date
FROM Projects
WHERE DATE_FORMAT(start_date, '%Y') = '2023';





Q18) Use the MySQL-specific IFNULL function to list all Employee Names. If an employee has no manager, display "CEO" instead of NULL.

SELECT employee_name, IFNULL(CAST(manager_id AS CHAR), 'CEO') AS manager
FROM Employees;





Q19) Find the top 3 Departments based on the total hours worked by all their employees across all projects.

SELECT d.dept_name, SUM(a.hours_worked) AS total_hours
FROM Departments d
JOIN Employees e ON d.dept_id = e.dept_id
JOIN Assignments a ON e.employee_id = a.employee_id
GROUP BY d.dept_name
ORDER BY total_hours DESC
LIMIT 3;





Q20) Display the Project Name and a status ("High Activity" if total hours > 300, "Normal" otherwise) using the CASE statement.

SELECT project_name,
    SUM(a.hours_worked) AS total_hours,
    CASE
        WHEN SUM(a.hours_worked) > 300 THEN 'High Activity'
        ELSE 'Normal'
    END AS status
FROM Projects p
JOIN Assignments a ON p.project_id = a.project_id
GROUP BY p.project_name;





Q21) List the Employee Name and their Manager's Name, even if the employee has no manager, or if a potential manager has no employees reporting to them.

SELECT e.employee_name, IFNULL(m.employee_name, 'No Manager') AS manager_name
FROM Employees e
LEFT JOIN Employees m ON e.manager_id = m.employee_id;





Q22) Write a query using a Common Table Expression (CTE) to find the Project Name and the number of unique employees assigned to it.

WITH ProjectEmployees AS (
    SELECT p.project_name, COUNT(DISTINCT a.employee_id) AS unique_employees
    FROM Projects p
    JOIN Assignments a ON p.project_id = a.project_id
    GROUP BY p.project_name
)
SELECT * FROM ProjectEmployees;





Q23) Find the Employee Name(s) who are assigned to the same projects as Employee ID 101.

SELECT DISTINCT e.employee_name
FROM Employees e
JOIN Assignments a ON e.employee_id = a.employee_id
WHERE a.project_id IN (
    SELECT project_id FROM Assignments WHERE employee_id = 101
)
AND e.employee_id != 101;
