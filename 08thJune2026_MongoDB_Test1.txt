QUESTIONS AND QUERIES:

Q1. Find all students from the CS department with marks greater than 80.

db.students.find({
    dept: "CS",
    marks: { $gt: 80 }
})


Q2. Find students whose marks are between 60 and 85 (inclusive) using $gte and $lte.

db.students.find({
    marks: {
        $gte: 60,
        $lte: 85
    }
})


Q3. Find all students who are either from IT or ECE department using $in operator.

db.students.find({
    dept: {
        $in: ["IT", "ECE"]
    }
})


Q4. Find students NOT in the CS department using $nin and project only name and dept

db.students.find(
    {
        dept: {
            $nin: ["CS"]
        }
    },
    {
        _id: 0,
        name: 1,
        dept: 1
    }
)


Q5. Find students where the address.city is "Chennai" using dot notation.

db.students.find({
    "address.city": "Chennai"
})



Q6. Find total revenue per category using $group and $sum on a computed field (price × sold).

db.products.aggregate([
    {
        $group: {
            _id: "$category",
            totalRevenue: {
                $sum: {
                    $multiply: ["$price", "$sold"]
                }
            }
        }
    }
])

Q7. Use aggregation to find total number of students and average marks per department.

db.students.aggregate([
    {
        $group: {
            _id: "$dept",
            totalStudents: { $sum: 1 },
            averageMarks: { $avg: "$marks" }
        }
    }
])

Q8. Remove the 'scholarship' field from all students using $unset.

db.students.updateMany(
    {},
    {
        $unset: {
            scholarship: ""
        }
    }
)


Q9. Add a new field 'scholarship' with value false to all CS students using $set.

db.students.updateMany(
    {
        dept: "CS"
    },
    {
        $set: {
            scholarship: false
        }
    }
)


Q10. Increment marks of student with _id:6 by 10 using $inc.

db.students.updateOne(
    { _id: 6 },
    {
        $inc: {
            marks: 10
        }
    }
)


Q11. Find the 3rd and 4th highest scoring students using sort, skip, and limit.

db.students.find()
    .sort({ marks: -1 })
    .skip(2)
    .limit(2)


Q12. Find employees whose salary is greater than 50000 and experience greater than 5.

db.employees.find({
    salary: { $gt: 50000 },
    experience: { $gt: 5 }
})


Q13. Find employees whose skills array size is greater than 2. 

db.employees.find({
    $expr: {
        $gt: [
            { $size: "$skills" },
            2
        ]
    }
})


Q14. Find departments whose avergae salary is greater than 50000.

db.employees.aggregate([
    {
        $group: {
            _id: "$department",
            averageSalary: { $avg: "$salary" }
        }
    },
    {
        $match: {
            averageSalary: { $gt: 50000 }
        }
    }
])


Q15. Find minimum salary in each department.

db.employees.aggregate([
    {
        $group: {
            _id: "$department",
            minimumSalary: { $min: "$salary" }
        }
    }
])

Q16. Rename field city to location.

//if for students table
db.students.updateMany(
    {},
    {
        $rename: {
            "address.city": "address.location"
        }
    }
)


//if for employee table
db.employees.updateMany(
    {},
    {
        $rename: {
            "city": "location"
        }
    }
)


Q17. Find products where stock is of type Number using $type operator.

db.products.find({
    stock: {
        $type: "int"
    }
})


Q18. Set the active status to true for all inactive students using updateMany and $set.

db.students.updateMany(
    {
        active: false
    },
    {
        $set: {
            active: true
        }
